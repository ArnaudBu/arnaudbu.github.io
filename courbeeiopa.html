Courbe des taux & Cohérence

Vendredi 12 mai 2017

1 premiere partie

# Chargement des librairies et des donn&#233;es

library(data.table)
library(ggplot2)
library(viridis)
library(plotly)
library(reshape2)
library(dplyr)

courbeEIOPA <- fread("doc/data/courbeeiopa/eiopa.csv", dec = ",")
tauxUS <- fread("doc/data/courbeeiopa/tauxUS.csv", dec = ",")
courbeEIOPA <- courbeEIOPA %>%
  tbl_df() %>%
  mutate(pzc2015 = exp(- Annees * Year2015),
         pzc2016 = exp(- Annees * Year2016)) %>%
  data.table()

courbeEIOPA$pzc2015as2016 <- NA
courbeEIOPA$pzc2015as2016[1:(nrow(courbeEIOPA)-1)] <-
  courbeEIOPA$pzc2015[2:nrow(courbeEIOPA)] / courbeEIOPA$pzc2015[1]

courbeEIOPA <- courbeEIOPA %>%
  tbl_df() %>%
  mutate(Year2015as2016 = -log(pzc2015as2016)/ Annees ) %>%
  data.table()

courbeEIOPA$fwd2015 <- NA
courbeEIOPA$fwd2015[1] <- courbeEIOPA$Year2015[1]
courbeEIOPA$fwd2015[2:nrow(courbeEIOPA)] <- 
  -log(courbeEIOPA$pzc2015[2:nrow(courbeEIOPA)] / courbeEIOPA$pzc2015[1:(nrow(courbeEIOPA)-1)])

courbeEIOPA$fwd2016 <- NA
courbeEIOPA$fwd2016[2] <- courbeEIOPA$Year2016[1]
courbeEIOPA$fwd2016[3:nrow(courbeEIOPA)] <- 
  -log(courbeEIOPA$pzc2016[2:(nrow(courbeEIOPA)-1)] / courbeEIOPA$pzc2016[1:(nrow(courbeEIOPA)-2)])
eiopaplot <- courbeEIOPA %>%
  tbl_df() %>%
  mutate(Annees = Annees + 2016) %>%
  select(Annees, Year2015as2016, Year2016) %>%
  data.table()

eiopaplot <- melt(eiopaplot, id.vars = "Annees")

levels(eiopaplot$variable) <- c("2015", "2016")

p <- ggplot(data = eiopaplot, aes(x = Annees, y = value, color = variable)) +
  geom_line() +
  theme_minimal() +
  scale_y_continuous(labels = scales::percent) +
  scale_color_viridis(discrete = TRUE, name = "")+
  xlab("Ann&#233;e d'observation") +
  ylab("Taux")

ggplotly(p)
eiopaplot <- courbeEIOPA %>%
  tbl_df() %>%
  mutate(Annees = Annees + 2014) %>%
  select(Annees, fwd2015, fwd2016) %>%
  data.table()

eiopaplot <- melt(eiopaplot, id.vars = "Annees")

levels(eiopaplot$variable) <- c("2015", "2016")

colnames(eiopaplot) <- c("Date", "Annee_calcul", "valeur")

spot <- eiopaplot %>%
  tbl_df() %>%
  filter(Date == Annee_calcul) %>%
  data.table()

p <- ggplot() +
  geom_line(data = eiopaplot, aes(x = Date, y = valeur, group = Annee_calcul), color = 'grey') +
  geom_line(data = spot, aes(x = Date, y = valeur), color = 'black')+
  theme_minimal() +
  scale_y_continuous(labels = scales::percent) +
  theme(legend.position="none") +
  xlab("Ann&#233;e d'observation") +
  ylab("Taux &#224; 1 an")

ggplotly(p)
tauxUS <- melt(tauxUS, id.vars = "Date")

colnames(tauxUS) <- c("Date", "Annee_calcul", "valeur")

p <- ggplot() +
  geom_line(data = tauxUS[Annee_calcul != "SPOT"], aes(x = Date, y = valeur, group = Annee_calcul), color = 'grey') +
  geom_line(data = tauxUS[Annee_calcul == "SPOT"], aes(x = Date, y = valeur), color = 'black')+
  theme_minimal() +
  scale_y_continuous(labels = scales::percent) +
  theme(legend.position="none") +
  xlab("Ann&#233;e d'observation") +
  ylab("Taux &#224; 1 an")

ggplotly(p)
